--only executed if the base part is CanCollide true,

local Clip = false
local Noclipping = nil
local speaker = game:GetService("Players").LocalPlayer

local function getplrs(s)
sl = s:lower()
if sl == "others" then
for i,v in pairs(game.Players:GetPlayers() do
if v ~= game.Players.LocalPlayer then
return v
end
end
end
end

 function enable()
	Clip = false
	wait(0.1)
	local function NoclipLoop()
		if Clip == false and getplrs("others").Character ~= nil then
			for _, child in pairs(getplrs("others").Character:GetDescendants()) do
				if child:IsA("BasePart") and child.CanCollide == true then
					child.CanCollide = false
				end
			end
		end
	end
	Noclipping = RunService.Stepped:Connect(NoclipLoop)
end

 function disable()
	if Noclipping then
		Noclipping:Disconnect()
	end
	Clip = true
end

local library = loadstring(game:HttpGet("https://raw.githubusercontent.com/bloodball/-back-ups-for-libs/main/turtle"))()

local Window = library:Window("Anti Fling by GGH52Lan")

local dropdown = Window:Dropdown("Mode", {"Enable", "Disable"}, function(o)
    if o == "Enable" then
     enable()
     if Clip == false then
     game:GetService("StarterGui"):SetCore("SendNotification",{
 Title = "GGH52Lan";
 Text = "Noclipping enable";
 Duration = 5;
 Icon = "rbxassetid://14260295451";
})
end
     else
     disable()
     if Clip == true then
     game:GetService("StarterGui"):SetCore("SendNotification",{
 Title = "GGH52Lan";
 Text = "Noclipping disable";
 Duration = 5;
 Icon = "rbxassetid://14260295451";
})
end
    end
end)
